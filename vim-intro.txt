[leader] - ,

navigation
  [leader]
    n - nerd tree
    a - search in files
    ff - fullscreen search in all files
    fl - search in buffers
    m - most recent files

buffers
  [C-n]  - next buffer
  [C-p]  - previous buffer
  [C-w]o - zoom window
  [leader]
    bl/l  - buffer list
    bo - only buffer only
    bd/q - kill current buffer
    bn - next buffer next
    bp - previous buffer

tests
  [leader]
    tn - test nearest
    tf - test file
    ta - test suite
    tl - test last
    tv - test visit

spotify
  [leader]
    sn - next song
    sb - previous song
    sp - play/pause

refactoring
  [leader]
    rs - change (:symbol =>) to (symbol:)
    w  - fixing whitespaces

git
  [leader]
    gc - git commit
